centos修改dhcpd日志不记入系统日志

编辑dhcp服务配置文件

vim /etc/dhcp/dhcpd.conf&nbsp;&nbsp; 在末尾加入

log-facility&nbsp; local4;

&nbsp;&nbsp;&nbsp; &nbsp;

编辑 rsyslog服务配置文件，修改/var/log/messages加入local4.none&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; /var/log/messages

vim /etc/rsyslog.conf&nbsp;&nbsp; 加入

# Save dhcpd messages also to dhcpd.log

local4.*&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; /var/log/dhcpd.log

#*.info;mail.none;authpriv.none;cron.none&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; /var/log/messages

*.info;mail.none;authpriv.none;cron.none;local4.none&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; /var/log/messages

重启  rsyslog 服务

service rsyslog restart

重启 dhcpd 服务

service dhcpd restart

tail -f /var/log/dhcpd.conf

tail -f /var/log/messges
